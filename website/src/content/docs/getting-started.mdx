---
title: Getting Started
description: Installation and minimal setup.
---

import { Tabs, TabItem } from '@astrojs/starlight/components';

Install the package.

<Tabs syncKey="pkg">
<TabItem label="pnpm">
```sh
pnpm add astro-image-processor
```
</TabItem>
<TabItem label="Bun">
```sh
Bun add astro-image-processor
```
</TabItem>
<TabItem label="Yarn">
```sh
yarn add astro-image-processor
```
</TabItem>
<TabItem label="npm">
```sh
npm install astro-image-processor
```
</TabItem>
</Tabs>

Add to the Astro settings.

```ts ins={3,7}
// astro.config.ts
import { defineConfig } from "astro/config";
import { astroImageProcessor } from "astro-image-processor";

export default defineConfig({
  integrations: [
    astroImageProcessor(),
  ],
});
```

Place global styles.

```astro ins={3,18}
// src/layout/Layout.astro
---
import { GlobalStyles } from 'astro-image-processor/components';
---

<!doctype html>
<html>
  <head>
    <meta charset="utf-8">
    <title>Astro Image Processor</title>
  </head>
  <body>
    <h1>Astro Image Processor</h1>
    <slot />
  </body>
</html>

<GlobalStyles />
```

Place component to the page.

```astro ins={3,8}
// src/pages/index.astro
---
import { Picture } from 'astro-image-processor/components';
import { Layout } from '../layout/Layout.astro';
---

<Layout>
  <Picture src="" alt="Sample image" />
</Layout>
```

Run build.

<Tabs syncKey="pkg">
<TabItem label="pnpm">
```sh
pnpm build
```
</TabItem>
<TabItem label="Bun">
```sh
bun run build
```
</TabItem>
<TabItem label="Yarn">
```sh
yarn build
```
</TabItem>
<TabItem label="npm">
```sh
npm build
```
</TabItem>
</Tabs>
